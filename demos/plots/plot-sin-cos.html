Simple Canvas example

green = SIN(x), yellow = COS(x)


    window.onload = function () {
        // get canvas and context
        var canvas = document.getElementById("myCanvas");
        var ctx = canvas.getContext("2d");

        drawAxes(canvas, ctx);
        let frq = 32;
        let amp = 40;

        ctx.fillStyle = "green";
        for (var x = 0; x < 180; x++) {
            var y =  amp * Math.sin(x / frq * 3.1415);
            plot(x, y, ctx);
        }

        ctx.fillStyle = "yellow";
        for (var x = 0; x < 180; x++) {
            var y =  amp * Math.cos(x / frq * 3.1415);
            plot(x, y, ctx);
        }
    };

    function plot(x, y, ctx) {
        let dx = 50;
        let dy = 100;
        let es = 2.5;
        ctx.fillRect((dx + x) * es, (dy - y) * es, 2, 2);
    }
