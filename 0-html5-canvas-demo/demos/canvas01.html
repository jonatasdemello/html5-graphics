Simple Canvas example

var canvas = document.getElementById("myCanvas"); var ctx = canvas.getContext("2d"); // Draw some rectangles with different color values ctx.fillStyle = "red"; ctx.fillRect(5, 5, 200, 200); ctx.fillStyle = "rgba(0,50,200,0.5)"; ctx.fillRect(55, 55, 200, 200); ctx.fillStyle = "rgba(0,255,10,0.5)"; ctx.fillRect(100, 100, 200, 200); ctx.stroke();
